[docs]    @classmethod
    def from_device(cls, device, parent=None, **kwargs):
        """
        Create a new instance of the tool for a Device

        Shortcut for:

        .. code::

            tool = TyphonTool()
            tool.add_device(device)

        Parameters
        ----------
        device: ophyd.Device

        parent: QWidget
        """
        instance = cls(parent=parent, **kwargs)
        instance.add_device(device)
        return instance
